require('dotenv').config(); const express = require('express'); const cors = require('cors'); const fs = require('fs'); const path = require('path'); const { Client, GatewayIntentBits, EmbedBuilder, ActionRowBuilder, ButtonBuilder, ButtonStyle, PermissionsBitField, } = require('discord.js'); /* ============ تخزين الطلبات (ملف JSON بسيط) ============ */ const DATA_FILE = path.join(__dirname, 'applications.json'); function loadApps() { if (!fs.existsSync(DATA_FILE)) fs.writeFileSync(DATA_FILE, '{}'); return JSON.parse(fs.readFileSync(DATA_FILE, 'utf8')); } function saveApps(data) { fs.writeFileSync(DATA_FILE, JSON.stringify(data, null, 2)); } /* ============ بوت الديسكورد ============ */ const client = new Client({ intents: [GatewayIntentBits.Guilds, GatewayIntentBits.GuildMembers], }); client.once('ready', () => { console.log(`✅ Bot logged in as ${client.user.tag}`); }); function buildEmbed(app, statusText, color) { return new EmbedBuilder() .setTitle('🛡️ طلب انضمام جديد / New Cadet Application') .setColor(color) .addFields( { name: '👤 Discord', value: `<@${app.discordId}>`, inline: true }, { name: '🆔 Discord ID', value: app.discordId, inline: true }, { name: '📅 Applied At', value: app.appliedAt, inline: false }, { name: '👤 الاسم الكامل', value: app.name, inline: true }, { name: '🎂 العمر', value: String(app.age), inline: true }, { name: '⏰ ساعات التواجد', value: String(app.hours), inline: true }, { name: '📄 الأدلة الصحيحة والغير صحيحة', value: app.evidence || 'لم يُجب' }, { name: '🎖️ أهم ثلاث بروتوكولات عسكرية', value: app.protocols || 'لم يُجب' }, { name: '💬 سبب الانضمام', value: app.why }, { name: '💼 الخبرة السابقة', value: app.exp || 'لا يوجد' }, { name: '📊 Status', value: statusText }, ) .setFooter({ text: 'Academy Application System' }) .setTimestamp(); } function fullRow(discordId) { return new ActionRowBuilder().addComponents( new ButtonBuilder().setCustomId(`accept_${discordId}`).setLabel('قبول').setStyle(ButtonStyle.Success).setEmoji('✅'), new ButtonBuilder().setCustomId(`reject_${discordId}`).setLabel('رفض').setStyle(ButtonStyle.Danger).setEmoji('❌'), new ButtonBuilder().setCustomId(`delete_${discordId}`).setLabel('حذف الطلب').setStyle(ButtonStyle.Secondary).setEmoji('🗑️'), ); } function deleteOnlyRow(discordId) { return new ActionRowBuilder().addComponents( new ButtonBuilder() .setCustomId(`delete_${discordId}`) .setLabel('حذف الطلب (للسماح بالتقديم مجددًا)') .setStyle(ButtonStyle.Secondary) .setEmoji('🗑️'), ); } /* ============ استقبال الطلبات من الموقع ============ */ const app = express(); app.use(cors()); app.use(express.json()); app.post('/apply', async (req, res) => { try { const { name, discordId, age, hours, evidence, protocols, why, exp } = req.body; if (!name || !discordId || !age || !hours || !why) { return res.status(400).json({ error: 'missing_fields' }); } if (!/^\d{15,22}$/.test(discordId)) { return res.status(400).json({ error: 'invalid_id' }); } const apps = loadApps(); if (apps[discordId]) { // موجود طلب سابق ولسا ما انحذف من قبل الإدارة return res.status(409).json({ error: 'already_applied' }); } const record = { name, discordId, age, hours, evidence: evidence || '', protocols: protocols || '', why, exp: exp || '', status: 'pending', appliedAt: new Date().toLocaleString('ar-SA', { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric', hour: '2-digit', minute: '2-digit', }), }; apps[discordId] = record; saveApps(apps); const channel = await client.channels.fetch(process.env.CHANNEL_ID); const embed = buildEmbed(record, '⏳ Pending Review', 0xd97a2a); await channel.send({ embeds: [embed], components: [fullRow(discordId)] }); res.json({ ok: true }); } catch (err) { console.error(err); res.status(500).json({ error: 'server_error' }); } }); app.get('/', (_req, res) => res.send('Academy Application Bot is running.')); /* ============ التعامل مع ضغطات الأزرار ============ */ client.on('interactionCreate', async (interaction) => { if (!interaction.isButton()) return; const [action, discordId] = interaction.customId.split('_'); if (!['accept', 'reject', 'delete'].includes(action)) return; const member = interaction.member; const isAdmin = member.permissions.has(PermissionsBitField.Flags.Administrator) || (process.env.ADMIN_ROLE_ID && member.roles.cache.has(process.env.ADMIN_ROLE_ID)); if (!isAdmin) { return interaction.reply({ content: '⛔ ليس لديك صلاحية لتنفيذ هذا الإجراء.', ephemeral: true }); } const apps = loadApps(); const record = apps[discordId]; if (action === 'delete') { delete apps[discordId]; saveApps(apps); if (record) { const embed = buildEmbed(record, '🗑️ تم حذف الطلب — يمكنه التقديم مجددًا', 0x6f8598); await interaction.update({ embeds: [embed], components: [] }); } else { await interaction.update({ components: [] }); } return; } if (!record) { return interaction.reply({ content: 'هذا الطلب لم يعد موجودًا (ربما انحذف مسبقًا).', ephemeral: true }); } if (action === 'accept') { record.status = 'accepted'; apps[discordId] = record; saveApps(apps); try { const user = await client.users.fetch(discordId); await user.send( `🎉 مبروك! تم قبول طلبك للانضمام لأكاديمية الشرطة.\n\n` + `رابط سيرفر الديسكورد: ${process.env.GUILD_INVITE_LINK}\n\n` + `بانتظارك للانضمام لأول حصة تدريبية.`, ); } catch (e) { console.log(`⚠️ تعذّر إرسال DM للمستخدم ${discordId} (الخاص مقفول عنده).`); } const embed = buildEmbed(record, '✅ Accepted', 0x93a565); await interaction.update({ embeds: [embed], components: [deleteOnlyRow(discordId)] }); return; } if (action === 'reject') { record.status = 'rejected'; apps[discordId] = record; saveApps(apps); const embed = buildEmbed(record, '❌ Rejected', 0xb3312c); await interaction.update({ embeds: [embed], components: [deleteOnlyRow(discordId)] }); return; } }); client.login(process.env.BOT_TOKEN); app.listen(process.env.PORT || 3000, () => { console.log(`🚀 API listening on port ${process.env.PORT || 3000}`); });
